# IPFS Configuration
# Copy this file to .env and fill in your values
# Option 1: Pinata (Recommended for production)
# Get your JWT token from https://app.pinata.cloud/
REACT_APP_PINATA_JWT=your_pinata_jwt_token_here
REACT_APP_PINATA_GATEWAY=https://gateway.pinata.cloud/ipfs/

# Option 2: Local IPFS Node
# Set to 'true' to use local IPFS node (requires IPFS daemon running)
REACT_APP_USE_LOCAL_IPFS=false
REACT_APP_LOCAL_IPFS_HOST=localhost
REACT_APP_LOCAL_IPFS_PORT=5001
REACT_APP_LOCAL_IPFS_PROTOCOL=http

# IPFS Gateway URLs (for retrieving data - fallback chain)
REACT_APP_IPFS_GATEWAY_1=https://gateway.pinata.cloud/ipfs/
REACT_APP_IPFS_GATEWAY_2=https://ipfs.io/ipfs/
REACT_APP_IPFS_GATEWAY_3=https://cloudflare-ipfs.com/ipfs/
